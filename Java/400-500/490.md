```java
class Solution {
    private int[] dx = new int[]{0, 0, 1, -1};
    private int[] dy = new int[]{1, -1, 0, 0};
    private boolean[][] visited;;
    public boolean hasPath(int[][] maze, int[] start, int[] destination) {
        int m = maze.length;
        int n = maze[0].length;
        visited = new boolean[m][n];
        Queue<int[]> q = new ArrayDeque<int[]>();
        q.offer(start);
        visited[start[0]][start[1]] = true;
        while (!q.isEmpty()) {
            int[] cur = q.poll();
            if (cur[0] == destination[0] && cur[1] == destination[1]) {
                return true;
            }
            for (int i = 0; i < 4; ++i) {
                int[] newPoint = rollingBall(cur, dx[i], dy[i], maze);
                if (visited[newPoint[0]][newPoint[1]] == false) {
                    q.offer(newPoint);
                    visited[newPoint[0]][newPoint[1]] = true;
                }
            }
        }
        return false;
    }
    public int[] rollingBall(int[] cur, int px, int py, int[][] maze) {
        int curX = cur[0];
        int curY = cur[1];
        int nx = curX + px;
        int ny = curY + py;
        while (check(maze, nx, ny)) {
            curX = nx;
            curY = ny;
            nx = curX + px;
            ny = curY + py;
        }
        return new int[]{curX, curY};
    }
    public boolean check(int[][] maze, int nx, int ny) {
        return (nx >= 0 && nx < maze.length && ny >= 0 && ny < maze[0].length && maze[nx][ny] != 1);
    }
}
```

Time complexity: O(m * n)  
Space complexity: O(m * n)
