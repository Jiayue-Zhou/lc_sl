```java
class Solution {
    public int[] searchRange(int[] nums, int target) {
        int n = nums.length;
        int left = 0;
        int right = n - 1;
        int leftIndex = -1;
        int rightIndex = -1;
        if (n == 0) {
            return new int[]{-1, -1};
        }
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] < target) {
                left = mid + 1;
            }
            else {
                right = mid;
            }
        }
        
        leftIndex = (nums[left] == target) ? left : -1;
        if (leftIndex == -1) {
            return new int[]{-1, -1};
        }
        left = 0;
        right = n - 1;
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] <= target) {
                left = mid + 1;
            }
            else {
                right = mid - 1;
            }
        }
        rightIndex = (nums[left] == target) ? left : left - 1;
        return new int[] {leftIndex, rightIndex};
    }
}
```
Time complexity: $O(logn)$  
Space complexity: O(1)
