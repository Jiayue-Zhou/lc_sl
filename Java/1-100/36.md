```java
class Solution {
    public boolean isValidSudoku(char[][] board) {
        for (int i = 0; i < 9; ++i) {
            for (int j = 0; j < 9; ++j) {
                char cur = board[i][j];
                if (cur == '.') {
                    continue;
                }
                else {
                    if (!isValid(board, i, j)) {
                        return false;
                    }
                }
            }
        }
        return true;
    }
    boolean isValid(char[][] board, int x, int y) {
        for (int i = 0; i < 9; ++i) {
            if (i != x && board[i][y] == board[x][y]) {
                return false;
            }
            if (i != y && board[x][i] == board[x][y]) {
                return false;
            }
        }
        int ix = x / 3 * 3;
        int iy = y / 3 * 3;
        for (int i = 0; i < 9; ++i) {
            int nx = ix + i / 3;
            int ny = iy + i % 3;
            if (board[x][y] == board[nx][ny] && (nx != x || ny != y)) {
                return false;
            }
        }
        return true;
    }
}
```

Time complexity: O(1)
Space complexity: O(1)
