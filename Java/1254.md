```java
class Solution {
    int[] dx = new int[]{0, 0, -1, 1};
    int[] dy = new int[]{1, -1, 0, 0};
    public int closedIsland(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int cnt = 0;
        for (int i = 0; i < m; ++i) {
            if (grid[i][0] == 0) {
                dfs(grid, i, 0);
            }
            if (grid[i][n - 1] == 0) {
                dfs(grid, i, n - 1);
            }
        }
        for (int i = 0; i < n; ++i) {
            if (grid[0][i] == 0) {
                dfs(grid, 0, i);
            }
            if (grid[m - 1][i] == 0) {
                dfs(grid, m - 1, i);
            }
        }
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == 0) {
                    dfs(grid, i, j);
                    cnt++;
                }
            }
        }
        return cnt;
    }
    public void dfs(int[][] grid, int x, int y) {
        grid[x][y] = 1;
        for (int i = 0; i < 4; ++i) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (check(grid, nx, ny)) {
                grid[nx][ny] = 1;
                dfs(grid, nx, ny);
            }
        }
    }
    public boolean check(int[][] grid, int x, int y) {
        return (x >= 0 && x < grid.length && y >= 0 && y < grid[0].length && grid[x][y] == 0);
    }
}

```

Time complexity: O(m * n)  
Space complexity: O(1)
