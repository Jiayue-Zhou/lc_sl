```java
class Solution {
    public int evalRPN(String[] tokens) {
        Deque<Integer> stk = new ArrayDeque<Integer>();
        int n = tokens.length;
        for (int i = 0; i < n; ++i) {
            int num1;
            int num2;
            if (tokens[i].equals("+")) {
                num2 = stk.pop();
                num1 = stk.pop();
                int num3 = num1 + num2;
                stk.push(num3);
            }
            else if (tokens[i].equals("-")) {
                num2 = stk.pop();
                num1 = stk.pop();
                int num3 = num1 - num2;
                stk.push(num3);
            }
            else if (tokens[i].equals("*")) {
                num2 = stk.pop();
                num1 = stk.pop();
                int num3 = num1 * num2;
                stk.push(num3);
            }
            else if (tokens[i].equals("/")) {
                num2 = stk.pop();
                num1 = stk.pop();
                int num3 = num1 / num2;
                stk.push(num3);
            }
            else {
                stk.push(Integer.valueOf(tokens[i]));
            }
        }
        return stk.pop();
    }

}

```


Time complexity: O(n)  
Space complexity: O(n)
