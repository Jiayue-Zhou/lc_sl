```c++
class Solution {
public:
    int minAddToMakeValid(string s) {
        int leftPar = 0;
        int n = s.size();
        int move = 0;
        for (int i = 0; i < n; ++i) {
            if (s[i] == ')') {
                if (leftPar == 0) {
                    move++;
                }
                else {
                    leftPar--;   
                } 
            }
            else {
                leftPar++;
            }
        }
        return move + leftPar;
    }
};
```
