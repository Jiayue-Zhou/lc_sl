```c++
class Solution {
public:
    vector<int> dx = {-1, -1, -1, 1, 1, 1, 0, 0};
    vector<int> dy = {1,   0, -1, 1, 0, -1, 1, -1};
    int shortestPathBinaryMatrix(vector<vector<int>>& grid) {
        int n = grid.size();
        // bfs
        queue<pair<int, int>> q;
        if (grid[0][0] == 1) {
            return -1;
        }
        if (n == 1) {
            return 1;
        }
        q.push({0, 0});
        grid[0][0] = 1;
        int path = 1;
        while (!q.empty()) {
            int size = q.size();
            path++;
            for (int i = 0; i < size; ++i) {
                pair<int, int> cur = q.front();
                int x = cur.first;
                int y = cur.second;
            
                bool flag = putEightDir(grid, q, x, y, n);
                if (flag) {
                    return path;
                }
                q.pop();
            }

        }
        return -1;
    }
    bool putEightDir(vector<vector<int>>& grid, queue<pair<int, int>>& q,
        int x, int y, int n) {
        for (int i = 0; i < 8; ++i) {
            int nx = x + dx[i];
            int ny = y + dy[i];

            if (nx >= 0 && nx < n && ny >= 0 && ny < n && grid[nx][ny] == 0) {
                if (nx == n - 1 && ny == n - 1) {
                    return true;
                }
                q.push({nx, ny});
                grid[nx][ny] = 1;
            }
        }
        return false;
    }
};
```
