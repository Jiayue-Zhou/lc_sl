class Solution {
public:
    bool checkSubarraySum(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> preSum(n + 1, 0);
        unordered_map<int, int> map;
        map[0] = 0;
        for (int i = 1; i <= n; ++i) {
            preSum[i] = preSum[i - 1]  + nums[i - 1];
            if (!map.count(preSum[i] % k)) {
                map[preSum[i] % k] = i;
            }
            else {
                if (i - map[preSum[i] % k] >= 2) {
                    return true;
                }
            }
        }
        return false;   
    }
};
