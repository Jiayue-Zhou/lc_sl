'''c++
class Solution {
public:
    int findPoisonedDuration(vector<int>& timeSeries, int duration) {
        int endPoi = 0;
        int n = timeSeries.size();
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            if (endPoi > timeSeries[i]) {
                // minus the overlap
                ans -= (endPoi - timeSeries[i]);
            }
            endPoi = timeSeries[i] + duration;
            ans += duration;
        }
        return ans;
    }
};

```
  
Time complexity: O(n)  
Space complexity: O(1)
