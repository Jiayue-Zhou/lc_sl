class Solution {
public:
    bool validPalindrome(string s) {
        int n = s.size();
        int start = 0;
        int end = n - 1;
        int sen = valid(s, start, end);
        if (sen == -1) {
            return true;
        }
        else {
            if (valid(s, sen + 1, end - sen) == -1 ||
                valid(s, sen, end - sen - 1) == -1) {
                return true;
            }
        }
        return false;
    }
    int valid(string s, int start, int end) {
        while  (start <= end) {
            if (s[start] != s[end])  {
                return start;
            }
            start++;
            end--;
        }
        return -1;
    }
};
